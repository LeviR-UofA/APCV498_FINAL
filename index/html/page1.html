Centimeters to Feet Conversion Program in C.

This is a simple conversion program that converts centimeters to feet & inches repeating until 0 is typed to End.

            #include <stdio.h>
            // declarations are const because they should not change
            // these are set numbers used for calculations later in the program
            const float centimetersInAFoot = 30.48;
            const int inchesInAFoot = 12;
    
            int main()
            {
                // other Declarations
                float height, feet;
                // first prompt for user to input their height
                printf("Enter a height in centimeters: ");
                // scanf to read user input
                scanf("%f", &height);
                // if statement to end program if the number is negative, with a print statement
                if (height < 0) {
                    printf("You have entered a negative number, please input a positive number\n");
                    return 0;
                }
                // while statement to only continue while input is positive
                while (height > 0) {
                    feet = height / centimetersInAFoot;
                    // using .1f we can print up to the first decimal value of the value the user inputed
                    // I found this helpful to limit the float to any number you need
                    printf("%.1f cm = %d feet, %.1f inches\n",
                    // height is the number the user inputed
                    height,
                    // set feet to an integer to only get the whole number instead of a float which gets decimals
                    (int) feet,
                    // using the calculation of feet we can subtract it from the user input to only have the decimal left
                    // Then multiply it by inchesinafoot to get our inches amount
                    (feet - (int) feet) * inchesInAFoot);
                    // prompt to enter a height will continue until the user decides to quit
                    printf("Enter a height in centimeters (Type 0 to quit): ");
                    scanf("%f", &height);
                }
                // message to let the user know they quit the program
                printf("You have successfully quit the program\n");
    
                return 0;
            }
